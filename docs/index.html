Quick Start

Installation

# Clone repository
git clone https://github.com/gift-framework/g2-forge.git
cd g2-forge

# Install dependencies
pip install -r requirements-core.txt
pip install -e .

# Verify installation
python -c "import g2forge as g2; print(g2.__version__)"

Basic Usage

import g2forge as g2

# Create config for any topology - networks auto-size!
config = g2.create_k7_config(
    b2_m1=8, b3_m1=30,  # Building block 1
    b2_m2=7, b3_m2=30   # Building block 2
)
# Results in topology: b‚ÇÇ=15, b‚ÇÉ=61

# Create trainer (works on CPU or CUDA)
trainer = g2.training.Trainer(config, device='cuda')

# Train G‚ÇÇ metric (15k epochs ‚âà 2-6 hours on GPU)
results = trainer.train(num_epochs=15000)

# Validate geometry
print(f"Final torsion: {results['final_metrics']['torsion']:.2e}")
print(f"H¬≤ rank: {results['final_metrics']['rank_h2']}")
print(f"H¬≥ rank: {results['final_metrics']['rank_h3']}")

# Save trained model
trainer.save_checkpoint('g2_metric_b2_15_b3_61.pt')
